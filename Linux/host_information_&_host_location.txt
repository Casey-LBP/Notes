# HOST INFORMATION
##################
# print basic system information
uname

# print the systems hostname
hostname

# same as hostname -d
dnsdomainname

# same as hostname -y
domainname

# same as hostname -y
nisdomainname

# same as hostname -y
ypdomainname

# set and display network interface information
ip

# older command to display and set network interface information
ifconfig

# HOST LOCATION
###############
# lookup hostnames, ip addresses and dns info
host

# look up the registraints of internet domains
whois

# check if a remote host is reachable
ping

# view the network path of a remote host
tracerout
